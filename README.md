#### **Challenge 2 - What does this function return?**

```js
const n = function(a) {
```

#### **Challenge 3 - What does this function return?**

```js
const array = [];

function r(arr) {
	arr[3] = 'a';
	const n = [];
	for (const k in arr) {
		n.push(k);
	}
	n;
}

r(array);
```
